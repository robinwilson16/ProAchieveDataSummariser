SELECT
		NR.WB_HybridEndYearID,
		NR.PG_CollegeTypeID,
		NR.PG_WBLFundAgeGroupID,
		NR.PG_SSA1ID,
		NR.PG_SSA2ID,
		NR.PG_ProgTypeID,
		NR.PG_FrameworkID,
		BMLeave = SUM ( NR.BMLeave ),
		BMAchFrame = SUM ( NR.BMAchFrame ),
		BMAchFrameOverallLeave = ROUND ( CAST ( SUM ( NR.BMAchFrame ) AS FLOAT ) / CAST ( SUM ( NR.BMLeave ) AS FLOAT ), 4 ) * 100
		INTO #NatRate_ER_Overall
	FROM PG_NationalRates_APP_Fwrk_Overall NR
	GROUP BY
		NR.WB_HybridEndYearID,
		NR.PG_CollegeTypeID,
		NR.PG_WBLFundAgeGroupID,
		NR.PG_SSA1ID,
		NR.PG_SSA2ID,
		NR.PG_ProgTypeID,
		NR.PG_FrameworkID
