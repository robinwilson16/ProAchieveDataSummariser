SELECT DISTINCT
	CL.PG_StudentID
FROM CL_Midpoint CL
INNER JOIN vCL_MYS_RDS_Seln MYS 
	ON MYS.CL_MidpointID = CL.CL_MidpointID
	AND MYS.PG_ProviderID = CL.PG_ProviderID
	AND MYS.DefaultSummary = 0
	AND MYS.IsArchived = 0
	AND MYS.IsQSRSummary = 0
	AND MYS.RulesApplied = 0
	AND MYS.IncludeAllAimTypes = 1
WHERE
	CL.PG_AcademicYearID = '17/18'

UNION

SELECT DISTINCT
	ER.PG_StudentID
FROM ER_Midpoint ER
INNER JOIN vER_MYS_RDS_Seln MYS 
	ON MYS.ER_MidpointID = ER.ER_MidpointID
	AND MYS.PG_ProviderID = ER.PG_ProviderID
	AND MYS.DefaultSummary = 1
	AND MYS.IsArchived = 0
	AND MYS.IsQSRSummary = 0
WHERE
	ER.PG_AcademicYearID = '17/18'
